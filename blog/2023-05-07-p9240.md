## 参考代码

```cpp
#include <bits/stdc++.h>
using namespace std;

const int inf=0x3f3f3f3f;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	int n;
	cin>>n;
	int mn=0,mx=inf;
	while(n--)
	{
		int a,b;
		cin>>a>>b;
		mn=max(mn,a/(b+1)+1);
		mx=min(mx,a/b);
	}
	cout<<mn<<' '<<mx<<'\n';
	return 0;
}
```
