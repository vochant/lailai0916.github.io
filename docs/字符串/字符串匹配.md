# 字符串匹配

## RK

```c++
#include <bits/stdc++.h>
using namespace std;

using ull=unsigned long long;
const int base=131;
ull Pow(ull a,ull b)
{
	ull res=1;
	while(b)
	{
		if(b&1)res=res*a;
		a=a*a;
		b>>=1;
	}
	return res;
}
int RK(string s1,string s2)
{
	int n=s1.length(),m=s2.length();
	if(n<m)return -1;
	ull h1=0,h2=0,p=Pow(base,m-1);
	for(int i=0;i<m;i++)
	{
		h1=h1*base+s1[i];
		h2=h2*base+s2[i];
	}
	s1[n]='$';
	for(int i=0;i<=n-m;i++)
	{
		if(h1==h2&&s1.substr(i,m)==s2)return i;
		h1=(h1-s1[i]*p)*base+s1[i+m];
	}
	return -1;
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	string s1,s2;
	while(cin>>s1>>s2)
	{
		cout<<RK(s1,s2)+1<<'\n';
	}
	return 0;
}
```
