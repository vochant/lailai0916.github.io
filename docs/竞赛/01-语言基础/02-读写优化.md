# 读写优化

ios::sync_with_stdio(false);

## 快速读入

```cpp
int read()
{
	int x=0,f=1;char c=getchar();
	while(c<'0'||c>'9'){if(c=='-')f=-1;c=getchar();}
	while(c>='0'&&c<='9'){x=x*10+c-48;c=getchar();}
	return x*f;
}
```

## 快速输出

```cpp
void write(int x)
{
	if(x<0)putchar('-'),x=-x;
	if(x>9)write(x/10);
	putchar(x%10+48);
}
```

## 例题

:::info[[洛谷 P10815 【模板】快速读入](https://www.luogu.com.cn/problem/P10815)]

给定 $n$ 个整数 $a_i$，求出他们的和。

数据范围：$\left|a_i\right|\le n\le10^8$。

:::

```cpp
#include <bits/stdc++.h>
using namespace std;

int read()
{
	int x=0,f=1;char c=getchar_unlocked();
	while(c<'0'||c>'9'){if(c=='-')f=-1;c=getchar_unlocked();}
	while(c>='0'&&c<='9'){x=x*10+c-48;c=getchar_unlocked();}
	return x*f;
}
void write(int x)
{
	if(x<0)putchar('-'),x=-x;
	if(x>9)write(x/10);
	putchar(x%10+48);
}
int main()
{
    int n=read();
    int ans=0;
    for(int i=1;i<=n;i++)
    {
    	int x=read();
    	ans+=x;
    }
    write(ans);
    return 0;
}
```
